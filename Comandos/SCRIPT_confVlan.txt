#Criando Vlans
enable
config terminal
vlan 10
 name Faculty/Staff
 exit
vlan 20
 name Students
 exit
vlan 30
 name Guest(Default)
 exit
vlan 99
 name Management&Native
 exit
vlan 150
 name VOICE
 end

#Endereçamento das Vlans
config terminal
interface f0/11
 switchport mode access
 switchport access vlan 10
 exit
interface f0/18
 switchport mode access
 switchport access vlan 20
 exit
interface f0/06
 switchport mode access
 switchport access vlan 30
 end

#Endereçamento da Vlan Voice
config terminal
interface f0/11
 mls qos trust cos
 switchport voice vlan 150
 end
