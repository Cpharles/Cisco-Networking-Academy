-> HOSTNAME:
	Switch>_'enable'
	Switch#_'conf terminal'
	Switch(config)#_'hostname (coloque o nome)'

-> SENHAS:
	Switch(config)#_'line console 0'
	Switch(config-line)#_'password (coloque a senha)'
	Switch(config-line)#_'login'
	Switch(config-line)#_'exit'
	Switch(config)#_'service password-encryption'

-> Modo EXEC privilegiado:
	Switch(config)#_'enable secret (coloque a senha)'

-> BANNER:
	Switch(config)#_'banner motd # texto #'

-> (SVI) - VLan1:
	Switch(config)#_'interface vlan 1'
	Switch(config-if)#_'ip address (IP address) (Mask)'
	Switch(config-if)#_no shutdown
	Switch(config-if)#_'exit'

-> ACESSO TELNET:
	Switch(config)#_'line vty 0 15'
	Switch(config-line)#_'password (coloque a senha)'
	Switch(config-line)#_'login'
	Switch(config-line)#_'transport input telnet'
	Switch(config-line)#_'exit'
	Switch(config)#_'exit'

-> VERIFICAR que foi gravada na NVRAM:
	Switch#_'show running-config'

-> BACKUP NO NVRAM:
	Switch#_'copy running-config startup-config'
